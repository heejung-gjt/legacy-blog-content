1-1. setInterval()

일정한 시간 간격으로 작업을 수행하기 위해서 사용한다. clearinterval 함수를 사용하여 중지 할 수 있다

타이머 실습

    const TIME_SEC = 5;
    const timer = document.querySelector('.timer');
    const btn = document.querySelector('.btn');
    let started = false;
    let timers = undefined; 
 
-------------------------------------------------------------------------------- ⑤

    function reduceTime(time){
      const minute = Math.floor(time/60);
      const second = time%60;
      timer.innerText=`${minute}:${second}`;
    }
-------------------------------------------------------------------------------- ④


    function startTimer(){
      let leftTime =TIME_SEC;
      reduceTime(leftTime);  //맨처음 초가 표시됨

      timers = setInterval(()=>{
        if(leftTime <=0){
          clearInterval(timers);
          return;
        }
        reduceTime(--leftTime);
      },1000)
    }
-------------------------------------------------------------------------------- ③
    function stopTimer(){
      clearInterval(timers);
    }

-------------------------------------------------------------------------------- ②

    function stop(){
      btn.innerText='▶';
      stopTimer();
    }

    function start(){
      btn.innerText='■';
      startTimer();
    }

--------------------------------------------------------------------------------- ①

    btn.addEventListener('click',()=>{

      if(started){
        stop();
      }else{
        start();
      }
      started=!started;

    });


1. btn을 클릭했을때 started가 true인 경우 타이머를 중지시키는 함수 stop() 호출하고 started가 false인 경우 타이머를 실행시키는 함수 stop()을 호출한다 이때 started=!started인 이유는 button은 누를때마다 번갈아가면서 stop과 start가 호출되어야 하기 때문에 started에 true와 false를 번갈아 가면서 주기 위해서이다

2. 각각 stop(), start()일때 실행되는 기능이다 stop일 경우 타이머를 멈추어준다.

3. 이때 타이머를 멈추기 위해서 clearInterval을 작성해준다

4. start일 경우 타이머를 실행시키기위해서 startTimer() 함수를 호출해준다 해당 함수에 setInterval 함수를 호출하여 TIME_SEC초 만큼 반복시켜준다 startTimer()가 한번 실행이 되면 setInterval 함수는 leftTime 의 값이 0 보다 클때동안 반복된다

5. 타이머를 잴 때 초가 줄어드는 기능을 작성한 함수이다


2-1. matches()

참고 mdn : https://developer.mozilla.org/en-US/docs/Web/API/Element/matches

match() 메서드는 제공된 selectorString에 의해 요소가 선택되는지 확인한다. 즉, 요소가 선택자 인지 확인한다 

var result = element.matches(selectorString); //resutl는 boolean으로 값을 받는다


match()메서드를 사용한 예)


1.아래 코딩에서 box1과 item1은 각각 첫번째에 있는 div(검은색 상자)와 첫번째의 고슴도치 이미지를 가리킨다 box2와 item2는 두번째 아이템들을 가리킨다

2.addEventListener로 box1을 클릭했을때

3.함수 changeItem()이 실행된다. 이때 호출받은 event의 target을 변수 target에 저장한 후 target이 되는 변수가 .item1, 즉 class가 item1이라면 해당 태그를 제거해준다 반대로 target이 되는 변수가 box1이라면 새로운 고슴도치 img를 생성해준다

4.box2를 click하면 함수 changeScale함수가 실행된다 이때 target이 되는 변수가 item2라면 해당 태그의 scale를 증가시켜준다

<script>
const box1 = document.querySelector('.box1');
    const box2 = document.querySelector('.box2');
    const item1 = document.querySelector('.item1');
    const item2 = document.querySelector('.item2');
    let count = 0;  //box2 누를때마다 scale 값 증가시키기 위해 변수지정
------------------------------------------------------------------- ①
    box1.addEventListener('click',changeItem);
------------------------------------------------------------------- ②

    function changeItem(){
      const target = event.target;

      if (target.matches('.item1')){
        item1.remove();
      }else if(target.matches('.box1')){
        item1.setAttribute('class','item1');
        item1.setAttribute('src','img/고슴도치.png');
        box1.appendChild(item1)
      }
    }
------------------------------------------------------------------- ③

    box2.addEventListener('click', changeScale);

    function changeScale(){
     
      const target = event.target;
      ++count;
      if (target.matches('.item2')){
        item2.style.transform=`scale(1.${count})` ;
      }
    }

    </script>



미니실습 - 고슴도치를 전부 클릭하면 게임종료, 나무 선택시 고슴도치 +1씩 증가


1.게임을 시작할때 화면에 아이템이 랜덤한 위치로 출력되게 하기 위해서 함수 randomItem을 호출하였다

2.field에 있는 아이템 중 고슴도치 아이템을 클릭하면 해당 이미지가 제거되고 나무를 클릭하면 고슴도치 이미지가 +1씩 되게 게임 로직을 구현하였다

3.게임이 종료되면 pop-up–hide 인 팝업창을 visibility를 이용하여 화면에 출력한다

   const startBtn = document.querySelector('.start__btn');
   const field = document.querySelector('.field');
   const item_size = field.getBoundingClientRect();
   const tree=document.querySelector('.tree');
   const popup =document.querySelector('.pop-up');
   const popup_back = document.querySelector('.pop-up__refresh');
   const aniSound = new Audio('./sound/carrot_pull.mp3');
   const treeSound = new Audio('./sound/bug_pull.mp3');
   const ITEM_COUNT = 5;
   const TREE_COUNT = 10;
   const TREE_SIZE=150;
   let started=false;
   let count = 0;
   
   
   function init(){  //게임 시작할때마다 실행 
       field.innerHTML='';
       count=0;
       itemAdd('hedgehog',ITEM_COUNT,'img/고슴도치.png');  //itemAdd 함수에 생성하는 img의 name과 생성할 개수, img의 src를 매개변수로 보낸다
       itemAdd('tree',TREE_COUNT,'img/나무.png');
   }
   init();
   
   function itemAdd(name, count, src){  //아이템 생성하는 함수
       const x1 =0
       const y1=0
       const x2 = item_size.width-TREE_SIZE;
       const y2 = item_size.height-TREE_SIZE;
   
       for (let i=0; i<count; i++){
       const item = document.createElement('img');
       item.setAttribute('class', name);
       item.setAttribute('src', src);
       item.style.position='absolute';
       const random_x = randomItem(x1,x2);  //아이템 위치 random으로 생성
       const random_y = randomItem(y1,y2);
       item.style.left=`${random_x}px`;
       item.style.top=`${random_y}px`;
   
       field.appendChild(item)
       }
       
   }
   function randomItem(min, max){  //아이템 위치 랜덤으로 부여하기 위해 랜덤한 숫자 생성
       min = Math.ceil(min);
       max = Math.floor(max);
       return Math.floor(Math.random() * (max - min)) + min;
   }
-------------------------------------------------------------------------------------- ①
   
   field.addEventListener('click',onFieldClick);  //field에 있는 아이템 선택할때 
   
   function onFieldClick(){
       const target = event.target;
   
       if(target.matches('.hedgehog')){  //field에 있는 class이름이 hedgehog인 img 가 클릭되면
           target.remove();  //해당 이미지 제거
           count++;
           if(count == ITEM_COUNT){  //count한 개수가 아이템의 개수와 같다면
           finishGame();  //게임 종료
           }
           playSound(aniSound);  
       }else if(target.matches('.tree')){
           itemAdd('hedgehog',1,'img/고슴도치.png');
           count--;
           playSound(treeSound);
   
       }
   
       }
   function finishGame(){
           popup.classList.remove('pop-up--hide');
           startBtn.style.visibility='hidden';
           field.style.visibility='hidden';
   }
   
   function playSound(sound){
       sound.currentTime=0;
       sound.play();
   }

--------------------------------------------------------------------------------------  ② 

   
   popup_back.addEventListener('click',()=>{   //팝업창에 있는 ◀ 클릭하면
       startBtn.style.visibility='visible';
       popup.classList.add('pop-up--hide');
       startGame(); //다시 게임 시작
   });
   
   startBtn.addEventListener('click',()=>{  //startBtn 클릭하면

3-1.Audio()

참고 mdn : https://developer.mozilla.org/en-US/docs/Web/API/HTMLAudioElement/Audio

Audio 객체를 이용해 오디오를 재생할 수 있다. 자바스크립트에서 오디오 파일을 불러와 재생해준다

  let audio = new Audio('audio.mp3');
  audio.play();

------------------------------------------------------------------

// EX) new Audio로 오디오 파일을 불러와 해당 변수에 저장한다, playSound라는 함수로 play() 플레이 해준다. 
  
const aniSound = new Audio('./sound/carrot_pull.mp3');
const treeSound = new Audio('./sound/bug_pull.mp3');

playSound(aniSound);

function playSound(sound){
    sound.currentTime=0;   //다시 함수가 불러와질때 오디오가 처음부터 진행이 되도록 설정
    sound.play();
}
