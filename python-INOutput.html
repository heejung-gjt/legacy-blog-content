Python 입출력

1-1 input과 output

1.input

input을 사용하여 사용자에게 입력을 받을 수 있다. 이때 주의할 점은 input으로 출력하는 값들의 type은 모두 str이다

num = input('숫자입력 :')
#입력 : 3 -> 출력 : 3 , type : str로 출력


2.입력값을 int로 변경하고 싶을때는 형변환을 해주면 된다

def add(a, b):
    return a + b

num1 = int(input())  #int를 붙여 형변환
num2 = int(input())
result = add(num1, num2)
print(result)


3.end=’ ‘

한 줄에 결과값을 출력할때는 end=’ ‘를 사용한다. 이때 end=’ * ‘를 넣으면 문장과 문장 사이에는 *이 표시된다

num1 = int(input('숫자1입력 :'))
num2 = int(input('숫자2입력 :'))
print('num1의 숫자는 %d이다' % num1, end=' vs ')
print('num2의 숫자는 %d이다' % num2)

# 답 : num1의 숫자는 n이다 vs num2의 숫자는 n이다


4.ljust(), rjust()

ljust() 는 왼쪽으로 정렬을 할 때 사용한다 ljust(8)은 8칸을 확보하고 왼쪽부터 정렬을 시작하라는 의미이다

scores = {'수학' : 0, '영어' : 50}
for subject, score in scores.items():
    print(subject.ljust(8), str(score).rjust(4), sep=':')

'''
답  
수학      :   0
영어      :  50

'''

for num in range(1,16):

7.format

문자열 출력 방법 중 하나이다

# 1. 빈 자리는 빈공간으로 두고, 오른쪽 정렬을 하되, 총 10자리 공간을 확보
print(' {0: >10} '.format(500))
# 답 :        500

------------------------------------------------------------------------------------

# 2. 양수일땐 +로 표시 , 음수일땐 -로 표시
print('{0: >+10}'.format(500))
print('{0: >+10}'.format(-500))
'''
 답 :             +500
                  -500
'''

------------------------------------------------------------------------------------

# 3. 왼쪽 정렬하고, 빈칸으로 _로 채움
print('{0:_<+10}'.format(500))
#답 : +500______

------------------------------------------------------------------------------------

#4
#3자리마다 콤마를 찍어줌
print('{0:,}'.format(100000000000000000))

# #3자리마다 콤마를 찍어줌, +,-부호도 붙이기
print('{0:+,}'.format(100000000000000000))
print('{0:+,}'.format(-100000000000000000))

# #3자리 마다 콤마 찍어주기, 부호붙이기 자릿수 확인하기
# #빈자리는 ^로 채워주기
print('{0:^<+30,}'.format(100000000000000))

'''
답 :  100,000,000,000,000,000
     +100,000,000,000,000,000
     -100,000,000,000,000,000
     +100,000,000,000,000^^^^^^^^^^
'''

------------------------------------------------------------------------------------

#5
#소수점 출력
print('{0:f}'.format(5/3))

#소수점 특정 자리수 까지만 표시 (소수점 3째자리에서 반올림)
print('{0:.2f}'.format(5/3))

'''
답 :  1.666667
      1.67
'''

2-1 파일 입출력

1.파일을 생성하고 읽어올 수 있다

파일이 같은 경로에 생성된다. 생성후에는 close()로 파일을 닫아준다

  score_file = open('score.txt','w',encoding='utf8')  #한글 깨지지 않게 하기 위해 utf8을 써준다
  print('수학 : 90',file=score_file)
  print('영어 : 89',file=score_file)
  score_file.close()
   '''
    답 : score.txt 파일이 쓰기 형식으로 생성이 된다. 
    수학 : 90
    영어 : 89
    '''

#a는 append의 줄임말로써 기존에 있는 txt에 이어쓰고 싶을때 작성한다

score_file = open('score.txt','a',encoding='utf8') 
score_file.write('과학 : 93')
score_file.write('\n코딩 : 100')
score_file.close()
'''
답 : 수학 : 90
    영어 : 89
    과학 : 93
    코딩 : 100
'''


파일이 읽기 형식으로 불러와진다

#파일의 모든 내용 불러와 읽어준다
score_file = open('score.txt','r',encoding='utf8')  #r은 read를 뜻하는 것으로 파일 읽어오는 의미로 불러옴
print(score_file.read()) 
score_file.close()
'''
답 : 수학 : 90
    영어 : 89
    과학 : 93
    코딩 : 100
'''
------------------------------------------------------------------------------------

#한줄씩 읽어옴 , 줄별로 읽기, 한 줄 읽고 커서는 다음 줄로 이동
score_file =open('score.txt','r',encoding='utf8')
print(score_file.readline(),end=' ') #readline이 한줄씩 읽어오는 기능
print(score_file.readline()) 
'''
답 : 수학 : 90
    영어 : 89
'''
------------------------------------------------------------------------------------

# 몇줄인지 모를때 , 반복문을 통해서 불러올 수 있다
score_file =open('score.txt','r',encoding='utf8')
while True:
    line = score_file.readline()
    if not line:  #만약 line이 더이상 없다면
        print('\nNO LINE -- EXIT')
        break
    print(line, end='')
score_file.close()
'''
답 : 수학 : 90
    영어 : 89
    과학 : 93
    코딩 : 100
    NO LINE -- EXIT
'''
------------------------------------------------------------------------------------

# 리스트에 넣어서 처리 가능
score_file =open('score.txt','r',encoding='utf8')
lines = score_file.readlines()  #list 형태로 저장
for line in lines:
    print(line, end='')
print('\n',lines)
'''
답 : 수학 : 90
    영어 : 89
    과학 : 93
    코딩 : 100
     ['수학 : 90\n', '영어 : 89\n', '과학 : 93\n', '코딩 : 100']
'''


3-1 pickle

1.pickle은 프로그램 상에서 우리가 사용하고 있는 데이터를 파일형태로 저장하고 pickle을 이용해서 또다시 코드로 사용가능하다

파일이 같은 경로에 생성된다. 생성후에는 close()로 파일을 닫아준다

#profile.pickle 파일 생성

  import pickle
  profile_file = open('profile.pickle','wb') #b는 바이너리를 의미 피클 쓰기위해선 항상 b를 정의해주어야 한다
  profile ={'이름 ':'박명수','나이':30,'취미':['축구','골프']}
  print(profile)
  pickle.dump(profile,profile_file) #profile에 있는 정보를 profile_file에 저장한다
  profile_file.close() 
-----------------------------------------------------------------------------------

#profile.pickle 파일 불러오기

import pickle
profile_file=open('profile.pickle','rb')
profile = pickle.load(profile_file) #file에 있는 정보를 profile에 불러오기
print(profile)
profile_file.close()
'''
답 : {'이름 ': '박명수', '나이': 30, '취미': ['축구', '골프']}
'''


4-1 with

1.with를 이용해서 파일을 편하게 읽어오거나 불러올 수 있다

따로 close를 사용하지 않아도 된다

import pickle

with open('profile.pickle','rb') as profile_file:  
    print(pickle.load(profile_file))

2.with를 이용한 파일 읽기, 쓰기

import pickle

 #with이용한 파일 쓰기
with open('study.txt','w',encoding='utf8') as study_file:
    study_file.write('파이썬 열심히 공부중')     # study.txt 생성 

 #with이용한 파일 읽기
with open('study.txt','r',encoding='utf8')as study_file:
    print(study_file.read())                  # study.txt 내용 읽기
