import random

stay = 0
switch = 0
trial = 10000

for _ in range(trial):
    doors = [0,0,1]
    random.shuffle(doors)

    users_choice = doors.pop()

    doors.remove(0)

    if users_choice == 1:
        stay += 1
    else:
        switch += 1
print(f'stay : {stay/trial}, switch{switch/trial}')
