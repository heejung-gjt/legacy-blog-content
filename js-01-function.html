1-1. getBoundingClientRect()

참고 mdn

Element의 크기와 위치를 알고 싶을때 사용한다.

domRect = element.getBoundingClientRect();

위에 이미지에서 검은색 바탕으로 되어 있는 element의 크기를 구한 값이다

 const item = document.createElement('div');
        item.setAttribute('class','name');

위의 이미지에 별,하트,구름 아이템들을 javascript로 동적으로 넣어주었다. moveItem이라는 함수를 호출하여 총 4번 호출되게 하기 위해 매개변수에 작성하였다. i라는 태그를 html에 생성하고 class이름을 호출하여 받아온 값으로 한 뒤 star이라는 div 태그에 child로 집어 넣었다 . body안의 elements를 보면 star안에 총 12개의 아이콘들이 들어간 것을 볼 수 있다


function moveItem(name, count){
    for(let i=0 ; i<count ; i++){
        const item = document.createElement('i');
        item.setAttribute('class',name);
        star.appendChild(item);
    }
}

function init(){
    moveItem('fas fa-star',4,);
    moveItem('fas fa-heart',4);
    moveItem('fas fa-cloud',4);
}
init();



1-3. random()

참고 mdn - random


1.Math.random() 함수는 사용자가 원하는 랜덤한 범위를 만들 수 있다 두 값 사이의 난수 생성할 수 있다. 함수의 반환값은 min보다 크거나 같으며 max보다 작다

function getRandomArbitrary(min, max) {
  return Math.random() * (max - min) + min;
}


2.두 값 사이의 정수 난수 생성 할 수 있다. 함수의 반환값은 min보다 크거나 같으며 max보다 작다

function getRandomInt(min, max) {
  min = Math.ceil(min);
  max = Math.floor(max);
  return Math.floor(Math.random() * (max - min)) + min; //최댓값은 제외, 최솟값은 포함
}


동적으로 여러개 생성한 아이템들의 위치를 랜덤하게 배치하기 위해서 randomItem의 함수를 호출하여 원하는 값 사이의 난수를 생성하였다 star 박스안에서만(검은색 박스) 랜덤하게 움직이게 하기 위해서 star 박스의 width와 height을 각각 x2, y2에 넣어주었다. 이때 ITEM_SIZE는 아이템들의 사이즈이다. 아이템들의 크기만큼 빼주어야 star 박스밖으로 초과해서 나가지 않는다.


const star = document.querySelector('.star');
const star_wh = star.getBoundingClientRect();
const ITEM_SIZE = 50;

function randomItem(min, max){
    return Math.random() * (max - min) + min;
}

function moveItem(name, count){
    const x1=0; 
    const x2=star_wh.width-ITEM_SIZE; //star 박스의 width의 가장 끝값
    const y1=0; 
    const y2=star_wh.height-ITEM_SIZE; //star 박스의 height의 가장 끝값
    for(let i=0 ; i<count ; i++){  //한번씩 루틴이 돌기 때문에 모든 items들 각각 랜덤으로 배치 가능
        const item = document.createElement('i');
        item.setAttribute('class',name);
        item.style.position='absolute';
        const x = randomItem(x1,x2);  // width값 원하는 범위에서 랜덤하게 배치
        const y = randomItem(y1,y2);  // height 원하는 범위에서 랜덤하게 배치
        item.style.left=`${x}px`;  //left 값 랜덤한 x값으로 설정
        item.style.top=`${y}px`;   //top 값 랜덤한 y값으로 설정
        star.appendChild(item);
    }
}

START 버튼을 누르면 아이템들이 박스에 보이고 반복해서 누르면 아이템들이 랜덤으로 배치가 된다 addEventListener 함수를 이용하여 Btn을 click 하면 startMove() 함수를 출력한다 startMove 함수에서는 가장 먼저 init() 함수를 호출하여 아이템들을 호출해준다. 이때 star.innerHTML=’’ 을 해준 이유는 Btn을 클릭했을때 화면에 전부 지워진후 다시 아이템이 생성되어야 아이템이 무한으로 생기는 것을 방지할 수 있다. 그 후 star 박스의 css를 visibility=’visible’ 로 변경하여 화면에 있는 아이템들이 보일 수 있도록 한다


const Btn = document.querySelector('.start');
const star = document.querySelector('.star');

function startMove(){
    init();
    star.style.visibility='visible';
}

Btn.addEventListener('click',()=>{
    startMove();
});

function init(){
    star.innerHTML='';
    moveItem('fas fa-star',4,);
    moveItem('fas fa-heart',4);
    moveItem('fas fa-cloud',4);
}
init();

let moving = false;  //moving 변수는 값이 변하므로 let으로 설정한다

    if(moving){  //moving이 true일 경우
