from time import time

def time_checker(func):
    def wrapper(seq):
        start_at = time()
        result = func(seq)
        end_at = time()
        print('execution time: {}sec'.format(end_at-start_at))
        return result

    return wrapper

@time_checker
def fibo_rec(num):
    if num < 2:
        return num
    else:
        return fibo_rec(num-2) + fibo_rec(num-1)

def memoize(func):
    memo = {}
    def wrapper(seq):
        if seq not in memo:            
            memo[seq] = func(seq)
        return memo[seq]
    return wrapper
    
@memoize
def fib_memo(num):
    if num<2:
        return num
    else:
        return fib_memo(num-1) + fib_memo(num-2)
